#model evaluation: entropy, APPA, OCC

metrics_long_from_pprob <- function(pprob, model_name, conv = NA_integer_) {
  prob_cols <- grep("^prob\\d+$", names(pprob), value = TRUE)
  K <- length(prob_cols); stopifnot(K >= 2)

  P <- as.matrix(pprob[, prob_cols, drop = FALSE])
  P <- pmax(P, 1e-12)   # guards against log(0)
  n <- nrow(P)

  # Relative entropy (R) = 1 - normalized entropy (> 0.50)
  rel_entropy <- 1 + sum(P * log(P)) / (n * log(K))

  # Hard assignment + max posterior
  assign  <- max.col(P, ties.method = "first")
  maxprob <- P[cbind(seq_len(n), assign)]

  # APPA and OCC per class
  appa <- sapply(1:K, function(c) mean(maxprob[assign == c], na.rm = TRUE))
  occ  <- appa / ((1 - appa) / (K - 1))

  # Class proportions
  pct  <- sapply(1:K, function(c) mean(assign == c))

  tibble(
    model   = model_name,
    class   = factor(1:K, levels = 1:K),
    K       = K,
    n       = n,
    conv    = conv,
    entropy_rel = rel_entropy,  
    APPA    = appa,
    OCC     = occ,
    pct     = pct
  )
}
